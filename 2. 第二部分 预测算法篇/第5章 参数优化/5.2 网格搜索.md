```python
import pandas as pd
import numpy as np

iris = pd.read_csv("iris.csv")
x1 = iris.loc[:,['Sepal.Length']].values
x2 = iris.loc[:,['Sepal.Width']].values
x3 = iris.loc[:,['Petal.Length']].values
y = iris.loc[:,['Petal.Width']].values

minMSE=1000
k=55
for a in range(k+1):
    for b in range(k+1):
        for c in range(k+1):
            for d in range(k+1):
                a0 = 2.0*a/k -1
                b0 = 2.0*b/k -1
                c0 = 2.0*c/k -1
                d0 = 2.0*d/k -1
                y0 = a0 + b0*x1 + c0*x2 + d0*x3
                mse = np.mean((y-y0)**2)
                if mse < minMSE:
                    minMSE = mse
                    f_a0 = a0
                    f_b0 = b0
                    f_c0 = c0
                    f_d0 = d0

minMSE
```
